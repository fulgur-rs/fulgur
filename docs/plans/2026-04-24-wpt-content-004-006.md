**Step 3: Update `parse_selector_list` to return `Vec<ComplexSelector>` and handle `+`**

Replace the current `parse_selector_list` function entirely.

The new logic:

- Parse tokens as before, building a `CompoundSelector`
- When a `Token::Delim('+')` is seen: take the current compound as `prev_sibling_candidate`, reset to a fresh compound, set `saw_plus = true`
- On Comma or EOF: push a `ComplexSelector { subject: current, prev_sibling: prev_sibling_candidate }`
- Whitespace after `+` is allowed (skip it)
- Descendant combinator (whitespace between non-empty compounds without `+`) still returns empty → whole rule dropped

```rust
pub fn parse_selector_list(input: &str) -> Vec<ComplexSelector> {
    let mut parser_input = ParserInput::new(input);
    let mut parser = Parser::new(&mut parser_input);

    let mut selectors: Vec<ComplexSelector> = Vec::new();
    let mut current = CompoundSelector::default();
    let mut prev_sibling: Option<CompoundSelector> = None;
    let mut just_saw_whitespace = false;

    loop {
        let tok = match parser.next_including_whitespace() {
            Ok(t) => t.clone(),
            Err(_) => break,
        };

        if matches!(tok, Token::WhiteSpace(_)) {
            just_saw_whitespace = !current.parts.is_empty();
            continue;
        }

        match tok {
            Token::Comma => {
                if current.parts.is_empty() {
                    return Vec::new();
                }
                selectors.push(ComplexSelector { subject: std::mem::take(&mut current), prev_sibling: prev_sibling.take() });
                just_saw_whitespace = false;
            }
            Token::Delim('+') => {
                // Adjacent sibling combinator. Current compound becomes prev_sibling.
                if current.parts.is_empty() {
                    return Vec::new(); // `+` at start is invalid
                }
                prev_sibling = Some(std::mem::take(&mut current));
                just_saw_whitespace = false;
            }
            Token::Delim('.') => {
                if just_saw_whitespace && prev_sibling.is_none() {
                    // Descendant combinator — unsupported
                    return Vec::new();
                }
                // whitespace after `+` is allowed (just_saw_whitespace is from before `+`)
                match parser.next_including_whitespace() {
                    Ok(Token::Ident(name)) => {
                        current.parts.push(SimpleSelector::Class(name.as_ref().to_string()));
                    }
                    _ => return Vec::new(),
                }
                just_saw_whitespace = false;
            }
            Token::Delim('*') => {
                if just_saw_whitespace && prev_sibling.is_none() {
                    return Vec::new();
                }
                current.parts.push(SimpleSelector::Universal);
                just_saw_whitespace = false;
            }
            Token::IDHash(name) | Token::Hash(name) => {
                if just_saw_whitespace && prev_sibling.is_none() {
                    return Vec::new();
                }
                current.parts.push(SimpleSelector::Id(name.as_ref().to_string()));
                just_saw_whitespace = false;
            }
            Token::Ident(name) => {
                if just_saw_whitespace && prev_sibling.is_none() {
                    return Vec::new();
                }
                current.parts.push(SimpleSelector::Type(name.as_ref().to_ascii_lowercase()));
                just_saw_whitespace = false;
            }
            // Unsupported: `>`, `~`, `:`, `[`, and anything else
            _ => return Vec::new(),
        }
    }

    if !current.parts.is_empty() {
        selectors.push(ComplexSelector { subject: std::mem::take(&mut current), prev_sibling: prev_sibling.take() });
    } else if selectors.is_empty() {
        return Vec::new();
    }
    selectors
}
```

**Step 4: Rename `matches_node` to `matches_complex` and add sibling check**

The new function needs access to `doc` to look up the previous element sibling.

```rust
/// Match a [`ComplexSelector`] against a blitz DOM node. Checks subject
/// compound and, if present, the immediately preceding element sibling.
pub fn matches_complex(sel: &ComplexSelector, node: &blitz_dom::Node, doc: &blitz_html::HtmlDocument) -> bool {
    if !matches_compound(&sel.subject, node) {
        return false;
    }
    match &sel.prev_sibling {
        None => true,
        Some(sibling_sel) => {
            // Walk the parent's children to find the element immediately
            // before `node` in DOM order.
            let Some(parent_id) = node.parent else { return false };
            let Some(parent) = doc.get_node(parent_id) else { return false };
            let siblings = &parent.children;
            let Some(pos) = siblings.iter().position(|&id| id == node.id) else {
                return false;
            };
            // Walk backward to find the previous *element* sibling (skip text nodes).
            for &prev_id in siblings[..pos].iter().rev() {
                let Some(prev_node) = doc.get_node(prev_id) else { continue };
                if prev_node.element_data().is_some() {
                    return matches_compound(sibling_sel, prev_node);
                }
            }
            false
        }
    }
}
```

Also add a helper `matches_compound` (rename the old `matches_node` body):

```rust
fn matches_compound(sel: &CompoundSelector, node: &blitz_dom::Node) -> bool {
    let Some(elem) = node.element_data() else {
        return false;
    };
    if sel.parts.is_empty() {
        return false;
    }
    sel.parts.iter().all(|part| match part {
        SimpleSelector::Universal => true,
        SimpleSelector::Type(tag) => elem.name.local.as_ref().eq_ignore_ascii_case(tag),
        SimpleSelector::Class(want) => crate::blitz_adapter::get_attr(elem, "class")
            .is_some_and(|classes| classes.split_whitespace().any(|c| c == want)),
        SimpleSelector::Id(want) => {
            crate::blitz_adapter::get_attr(elem, "id").is_some_and(|id| id == want)
        }
    })
}
```

Keep `matches_node` as a compatibility shim if needed, or remove and update call sites.

fn parse_break_before_value<'i>(
    input: &mut Parser<'i, '_>,
) -> Result<crate::pageable::BreakBefore, ParseError<'i, ()>> {
    let ident = input.expect_ident()?.clone();
    match ident.as_ref().to_ascii_lowercase().as_str() {
        "page" | "always" | "left" | "right" | "recto" | "verso" => {
            Ok(crate::pageable::BreakBefore::Page)
        }
        "auto" => Ok(crate::pageable::BreakBefore::Auto),
        _ => Err(input.new_error(BasicParseErrorKind::QualifiedRuleInvalid)),
    }
}
